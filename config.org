** Auto-saving
I keep losing work, so let's make Emacs auto-save frequently. We'll use the `super-save` package to save buffers when we switch buffers or windows, plus an aggressive automatic save on idle or losing focus.
#+begin_src emacs-lisp
  (use-package super-save
    :ensure t
    :pin "melpa"
    :config
    ;; Save on window/buffer movements
    (add-to-list 'super-save-triggers 'other-window)
    (add-to-list 'super-save-triggers 'windmove-up)
    (add-to-list 'super-save-triggers 'windmove-down)
    (add-to-list 'super-save-triggers 'windmove-left)
    (add-to-list 'super-save-triggers 'windmove-right)
    
    (super-save-mode +1)
    (setq super-save-auto-save-when-idle t)
    (setq super-save-idle-duration 1)) ;; save after 1 second of idle

  ;; Also save unconditionally when Emacs loses focus
  (add-hook 'focus-out-hook (lambda () (save-some-buffers t)))
#+end_src

All it does is tell you what face is at point. It toggles hl-line-mode if needed.
#+BEGIN_SRC emacs-lisp
  (defun stag-what-face (pos)
    (interactive "d")
    (let* ((local-hl-p hl-line-mode)
           (global-hl-p global-hl-line-mode)
           (off (if local-hl-p (hl-line-mode -1)))
           (off (if global-hl-p (global-hl-line-mode -1)))
           (face (or (get-char-property pos 'read-face-name)
                     (get-char-property pos 'face))))
      (if face (message "Face: %s" face)
        (message "No face at %d" pos))
      (if local-hl-p (hl-line-mode))
      (if global-hl-p (global-hl-line-mode))))
#+END_SRC

#+BEGIN_SRC emacs-lisp
  ;;; org-gtd priority and tag inheritance for projects

  (defgroup my/org-gtd-inherit nil
    "Customization for org-gtd priority and tag inheritance."
    :group 'org-gtd
    :prefix "my/org-gtd-inherit-")

  (defcustom my/org-gtd-inherit-priority t
    "Whether to inherit priority from parent project.
When non-nil, project tasks automatically inherit the priority
from their parent project heading."
    :type 'boolean
    :group 'my/org-gtd-inherit)

  (defcustom my/org-gtd-inherit-tags t
    "Whether to inherit tags from parent project.
When non-nil, project tasks automatically inherit the tags
from their parent project heading."
    :type 'boolean
    :group 'my/org-gtd-inherit)

  (defun my/org-gtd-inherit--get-parent-metadata ()
    "Get priority and tags from parent heading.
Returns plist with :priority and :tags, or nil if no parent.
Handles buffer narrowing automatically."
    (save-excursion
      (save-restriction
        (widen)
        (when (org-up-heading-safe) ;; it's a dag so have to find the project heading for the org-gtd project property
          (list :priority (org-entry-get (point) "PRIORITY")
                :tags (org-get-tags nil t))))))

  (defun my/org-gtd-inherit--apply-metadata (metadata)
    "Apply METADATA plist (:priority and :tags) to current heading.
Only applies values that are configured for inheritance via customize vars."
    (when-let ((priority (and my/org-gtd-inherit-priority
                              (plist-get metadata :priority))))
      (org-priority (string-to-char priority)))
    (when-let ((tags (and my/org-gtd-inherit-tags
                          (plist-get metadata :tags))))
      (org-set-tags tags)))

  (defun my/org-gtd-inherit--prompt-for-tags ()
    "Prompt user for tags using standard org-mode interface."
    (call-interactively #'org-set-tags-command))

  (defun my/org-gtd-inherit-parent-metadata ()
    "Inherit priority and tags from parent when processing project tasks.

For project tasks, auto-copy priority and tags from parent heading.
For project headings and other item types, prompt for tags normally.
For someday items, skip tag prompting.

Inheritance respects `my/org-gtd-inherit-priority' and
`my/org-gtd-inherit-tags' customization variables."
    (cond
     ((org-gtd-organize-type-member-p '(project-task))
      ;; Project task: inherit from parent
      (when-let ((metadata (my/org-gtd-inherit--get-parent-metadata)))
        (my/org-gtd-inherit--apply-metadata metadata)))
     ((org-gtd-organize-type-member-p '(someday))
      ;; Someday: do nothing (skip tags)
      nil)
     (t
      ;; Everything else: prompt for tags
      (my/org-gtd-inherit--prompt-for-tags))))

  (defun tdw/agenda-move-to-someday ()
    "Move the task at point to Someday/Maybe.
If the task is already in Someday (e.g. after pressing c by mistake), resets
its TODO state in-place without refiling.  Strips any stale CNCL/NEXT markers.
Updates the agenda-line marker after refile so change-your-mind works."
    (interactive)
    (let ((marker (or (org-get-at-bol 'org-marker)
                      (org-agenda-error)))
          (heading nil)
          (task-id nil))
      ;; Backend: change properties and optionally refile
      (org-with-point-at marker
        (let ((gtd-type (org-entry-get (point) "ORG_GTD")))
          (setq heading (org-get-heading t t t t))
          (setq task-id (org-id-get (point) 'create))
          (if (string= gtd-type org-gtd-someday)
              ;; Already in Someday — just reset TODO state (undo stray CNCL/etc.)
              (progn
                (org-todo "")
                (save-excursion
                  (org-back-to-heading t)
                  (forward-line 1)
                  (when (looking-at "CLOSED: ")
                    (delete-region (point) (1+ (line-end-position)))))
                (save-buffer))
            ;; Coming from Actions — update ORG_GTD and refile
            (org-entry-put (point) "ORG_GTD" org-gtd-someday)
            (org-todo "")
            ;; Clear stale CLOSED timestamp (left behind by DONE/CNCL)
            (save-excursion
              (org-back-to-heading t)
              (forward-line 1)
              (when (looking-at "CLOSED: ")
                (delete-region (point) (1+ (line-end-position)))))
            (let ((org-gtd-organize-hooks nil))
              (let ((targets (org-gtd-refile--get-targets "Someday")))
                (if (null targets)
                    (user-error "No Someday refile targets found")
                  (org-refile nil nil (car targets)))))
            (save-buffer))))
      ;; Update marker to new location so change-your-mind works
      (when task-id
        (let ((new-pos (org-id-find task-id 'marker)))
          (when new-pos
            (let ((inhibit-read-only t))
              (put-text-property (line-beginning-position) (line-end-position)
                                 'org-marker new-pos)
              (put-text-property (line-beginning-position) (line-end-position)
                                 'org-hd-marker new-pos)))))
      ;; Visual: strip all state markers (incl. CNCL) then insert SOMEDAY
      (let ((inhibit-read-only t))
        (save-excursion
          (beginning-of-line)
          (when (re-search-forward "\\(SOMEDAY\\|NEXT\\|CNCL\\|INBOX\\) " (line-end-position) t)
            (replace-match "")))
        (save-excursion
          (beginning-of-line)
          (when (re-search-forward "— +" (line-end-position) t)
            (insert (propertize "SOMEDAY " 'face 'warning)))
          ;; Fallback: truncated prefix ends with '… '
          (unless (save-excursion (beginning-of-line) (re-search-forward "SOMEDAY " (line-end-position) t))
            (beginning-of-line)
            (when (re-search-forward "… " (line-end-position) t)
              (insert (propertize "SOMEDAY " 'face 'warning))))))
      (message "→ Someday: %s" heading)))

  (defun tdw/get-effort-minutes (effort-str)
    "Convert EFFORT-STR to minutes. Handles H:MM, NNmin, NNh, bare numbers."
    (cond
     ((null effort-str) 0)
     ;; H:MM format
     ((string-match "\\([0-9]+\\):\\([0-9]+\\)" effort-str)
      (+ (* 60 (string-to-number (match-string 1 effort-str)))
         (string-to-number (match-string 2 effort-str))))
     ;; NNmin format (e.g. "30min")
     ((string-match "\\([0-9]+\\)min" effort-str)
      (string-to-number (match-string 1 effort-str)))
     ;; NNh format (e.g. "2h")
     ((string-match "\\([0-9]+\\)h" effort-str)
      (* 60 (string-to-number (match-string 1 effort-str))))
     ;; Bare number (assume minutes)
     ((string-match "^[0-9]+$" effort-str)
      (string-to-number effort-str))
     ;; Fallback
     (t 0)))

  (defun tdw/format-minutes (minutes)
    "Format MINUTES as an H:MM string (e.g. 90 -> \"1:30\")."
    (format "%d:%02d" (/ minutes 60) (% minutes 60)))

  (defun tdw/get-subtree-effort-minutes ()
    "Sum effort of current entry and all its children in minutes.
Children in a DONE or CNCL state are excluded — finished work is not
remaining load."
    (let ((total 0))
      (save-excursion
        (org-back-to-heading t)
        (let ((effort (org-entry-get (point) "EFFORT")))
          (when effort (setq total (tdw/get-effort-minutes effort))))
        (let ((end (save-excursion (org-end-of-subtree t) (point))))
          (outline-next-heading)
          (while (and (< (point) end) (not (eobp)))
            (let ((effort (org-entry-get (point) "EFFORT")))
              (when (and effort
                         (not (member (org-get-todo-state) '("DONE" "CNCL"))))
                (setq total (+ total (or (tdw/get-effort-minutes effort) 0)))))
            (outline-next-heading))))
      total))

  (defun tdw/get-sanity-effort-totals ()
    "Return a list (TOTAL ON-FIRE BELLS DECK BIG-ROCKS OTHER UNEST CONSIDERED PAUSED) formatted as H:MM.
TOTAL excludes PAUSED — paused work is parked and does not count toward active load.
Also includes future calendar meetings (ORG_GTD=Calendar with timestamps after now)."
    (setq tdw--ef-p0 0 tdw--ef-bell 0 tdw--ef-deck 0
          tdw--ef-considered 0 tdw--ef-paused 0
          tdw--ef-big 0 tdw--ef-other 0
          tdw--ef-calendar 0 tdw--ef-unest 0
          tdw--ef-now (current-time)
          tdw--ef-today-end (encode-time 0 0 0
                                         (1+ (decoded-time-day (decode-time)))
                                         (decoded-time-month (decode-time))
                                         (decoded-time-year (decode-time))))
    (dolist (file (org-agenda-files))
      (let ((expanded-file (expand-file-name file)))
        (when (file-exists-p expanded-file)
          (with-current-buffer (find-file-noselect expanded-file)
            (org-with-wide-buffer
             (goto-char (point-min))
             (org-map-entries #'tdw/--effort-accumulate nil nil))))))
    (let ((fmt #'tdw/format-minutes))
      (list (funcall fmt (+ tdw--ef-p0 tdw--ef-bell tdw--ef-deck
                            tdw--ef-considered tdw--ef-big
                            tdw--ef-other tdw--ef-calendar))
            (funcall fmt tdw--ef-p0)
            (funcall fmt tdw--ef-bell)
            (funcall fmt tdw--ef-deck)
            (funcall fmt tdw--ef-big)
            (funcall fmt tdw--ef-other)
            tdw--ef-unest
            (funcall fmt tdw--ef-considered)
            (funcall fmt tdw--ef-paused))))

  (defun tdw/--effort-counted-p (todo gtd-val)
    "Non-nil if an entry with TODO state and GTD-VAL ORG_GTD would be counted."
    (if tdw--ef-target-gtd
        (and todo (not (member todo '("DONE" "CNCL")))
             (string= gtd-val tdw--ef-target-gtd))
      (and (member todo '("NEXT" "TODO"))
           (not (member gtd-val '("Inbox" "Someday" "Habit" "Delegated" "Projects"))))))

  (defun tdw/--effort-ancestor-counted-p ()
    "Non-nil if an ancestor of the entry at point is itself counted.
Counted ancestors already include this entry's effort via
`tdw/get-subtree-effort-minutes', so counting it again would double it."
    (save-excursion
      (let (found)
        (while (and (not found) (org-up-heading-safe))
          (when (tdw/--effort-counted-p (org-get-todo-state)
                                        (org-entry-get (point) "ORG_GTD"))
            (setq found t)))
        found)))

  (defun tdw/--effort-accumulate ()
    "Accumulate effort for one org entry into tdw--ef-* variables."
    (let ((todo (org-get-todo-state))
          (gtd-val (org-entry-get (point) "ORG_GTD"))
          (heading (org-get-heading t t t t)))
      (if (and (not tdw--ef-target-gtd) (string= gtd-val "Calendar"))
          ;; Skip DNS (Do Not Schedule) blocks — they're not real meetings
          (unless (and heading (string-match-p "\\bDNS\\b" heading))
          (let* ((effort-str (org-entry-get (point) "Effort"))
                 (effort-mins (when effort-str
                                (condition-case nil
                                    (round (org-duration-to-minutes effort-str))
                                  (error 0))))
                 (ts (org-entry-get (point) "TIMESTAMP"))
                 (ts-time (when ts (org-time-string-to-time ts))))
            (when (and effort-mins (> effort-mins 0)
                       ts-time
                       (time-less-p tdw--ef-now ts-time)
                       (time-less-p ts-time tdw--ef-today-end))
              (setq tdw--ef-calendar (+ tdw--ef-calendar effort-mins)))))
        (progn
        ;; Habits: count only if not yet completed today and not scheduled
        ;; for the future (next occurrence hasn't arrived yet).
        (when (and (not tdw--ef-target-gtd) (string= gtd-val "Habit"))
          (let* ((effort (or (tdw/get-subtree-effort-minutes) 0))
                 (last-repeat (org-entry-get (point) "LAST_REPEAT"))
                 (completed-today (and last-repeat
                                       (string-match-p (format-time-string "%Y-%m-%d" tdw--ef-now)
                                                       last-repeat)))
                 (sched-str (org-entry-get (point) "SCHEDULED"))
                 (sched-time (when sched-str
                               (condition-case nil
                                   (org-time-string-to-time sched-str)
                                 (error nil))))
                 (scheduled-future (and sched-time
                                        (not (time-less-p sched-time tdw--ef-today-end)))))
            (unless (or completed-today scheduled-future)
              (setq tdw--ef-other (+ tdw--ef-other effort)))))
        ;; Regular tasks. Skip entries whose ancestor is already counted —
        ;; the ancestor's subtree sum includes this entry's effort.
        (when (and (tdw/--effort-counted-p todo gtd-val)
                   (not (tdw/--effort-ancestor-counted-p)))
          (let ((effort (or (tdw/get-subtree-effort-minutes) 0)))
            (cond ((tdw/p0-p) (setq tdw--ef-p0 (+ tdw--ef-p0 effort)))
                  ((tdw/bells-ringing-p) (setq tdw--ef-bell (+ tdw--ef-bell effort)))
                  ((tdw/on-deck-p) (setq tdw--ef-deck (+ tdw--ef-deck effort)))
                  ((tdw/recently-considered-p) (setq tdw--ef-considered (+ tdw--ef-considered effort)))
                  ((tdw/paused-p) (setq tdw--ef-paused (+ tdw--ef-paused effort)))
                  ((tdw/big-rock-p) (setq tdw--ef-big (+ tdw--ef-big effort)))
                  ((tdw/unestimated-p) (cl-incf tdw--ef-unest))
                  (t (setq tdw--ef-other (+ tdw--ef-other effort))))))))))

  (defun tdw/round-up-15 (mins)
    "Round MINS up to the next multiple of 15.
A 20- or 25-minute meeting is really a 30-minute meeting."
    (if (and mins (> mins 0))
        (* 15 (ceiling mins 15))
      (or mins 0)))

  (defun tdw/get-schedule-effort-totals ()
    "Return (RITUAL-TOTAL SPENT-TOTAL REMAINING-TOTAL) as H:MM strings.
RITUAL-TOTAL sums the effort of the Daily Rituals habits (see
`tdw--ritual-habits').  SPENT-TOTAL sums today's past calendar meetings
(each rounded up to the next 15 minutes).  REMAINING-TOTAL sums today's
future calendar meetings (rounded up to 15 minutes) plus non-ritual
habits still due.  Habits already completed today and DNS calendar
blocks are excluded."
    (let ((ritual 0)
          (cal-past 0)
          (cal-future 0)
          (habit 0)
          (now (current-time))
          (today-start (encode-time 0 0 0
                                    (decoded-time-day (decode-time))
                                    (decoded-time-month (decode-time))
                                    (decoded-time-year (decode-time))))
          (today-end (encode-time 0 0 0
                                  (1+ (decoded-time-day (decode-time)))
                                  (decoded-time-month (decode-time))
                                  (decoded-time-year (decode-time)))))
      (dolist (file (org-agenda-files))
        (let ((expanded-file (expand-file-name file)))
          (when (file-exists-p expanded-file)
            (with-current-buffer (find-file-noselect expanded-file)
              (org-with-wide-buffer
               (goto-char (point-min))
               (org-map-entries
                (lambda ()
                  (let ((gtd-val (org-entry-get (point) "ORG_GTD"))
                        (heading (org-get-heading t t t t))
                        (done (org-entry-is-done-p)))
                    (cond
                     ;; Calendar meetings today, split into past (spent)
                     ;; and future (remaining), each rounded up to 15 min.
                     ((and (string= gtd-val "Calendar") (not done)
                           heading (not (string-match-p "\\bDNS\\b" heading)))
                      (let* ((ts (org-entry-get (point) "TIMESTAMP"))
                             (ts-time (when ts (org-time-string-to-time ts)))
                             (effort-str (org-entry-get (point) "Effort"))
                             (mins (when effort-str
                                     (condition-case nil
                                         (round (org-duration-to-minutes effort-str))
                                       (error 0)))))
                        (when (and mins (> mins 0) ts-time
                                   (not (time-less-p ts-time today-start))
                                   (time-less-p ts-time today-end))
                          (if (time-less-p ts-time now)
                              (setq cal-past (+ cal-past (tdw/round-up-15 mins)))
                            (setq cal-future (+ cal-future (tdw/round-up-15 mins)))))))
                     ;; Habits: split ritual habits from the rest.
                     ;; Only count habits scheduled today or earlier (overdue).
                     ;; Skip habits whose next occurrence is in the future.
                     ((string= (or (org-entry-get (point) "STYLE") "") "habit")
                      (unless done
                        (let* ((is-ritual (and heading
                                               (cl-some (lambda (n) (string-match-p n heading))
                                                        tdw--ritual-habits)))
                               (last-repeat (org-entry-get (point) "LAST_REPEAT"))
                               (completed-today (and last-repeat
                                                      (string-match-p (format-time-string "%Y-%m-%d" now)
                                                                      last-repeat)))
                               (sched-str (org-entry-get (point) "SCHEDULED"))
                               (sched-time (when sched-str
                                             (condition-case nil
                                                 (org-time-string-to-time sched-str)
                                               (error nil))))
                               (scheduled-future (and sched-time
                                                       (not (time-less-p sched-time today-end))))
                               (effort (or (tdw/get-subtree-effort-minutes) 0)))
                          (unless (or completed-today scheduled-future)
                            (if is-ritual
                                (setq ritual (+ ritual effort))
                              (setq habit (+ habit effort))))))))))
                nil nil))))))
      (let ((fmt #'tdw/format-minutes))
        (list (funcall fmt ritual)
              (funcall fmt cal-past)
              (funcall fmt (+ cal-future habit))))))

  (defun tdw/format-view-banner (label icon effort &optional show-effort)
    "Return a full-width, centered banner string for an agenda view.
LABEL is the view name (e.g. \"Inbox\"), ICON an emoji (e.g. \"📥\"),
EFFORT the total estimated effort string (e.g. \"2:15\").
Line 1 is a centered title flanked by `═' rules spanning the window width;
line 2 is the centered estimated-effort total, omitted when SHOW-EFFORT
is nil."
    (let* ((width (1- (let ((w (get-buffer-window (current-buffer) t)))
                        (if w (window-width w) 80))))
           (title (format "%s  %s VIEW" icon (upcase label)))
           (tw (string-width title))
           (lpad (max 1 (/ (- width tw 2) 2)))
           (rpad (max 1 (- width tw 2 lpad)))
           (line1 (concat (make-string lpad ?═) " " title " " (make-string rpad ?═)))
           (effline (format "Total Estimated Effort: %s" effort))
           (epad (max 0 (/ (- width (string-width effline)) 2)))
           (line2 (concat (make-string epad ?\s) effline)))
      (if show-effort
          (concat
           (propertize line1 'face '(:weight bold :foreground "#5fafff"))
           "\n"
           (propertize line2 'face '(:weight bold)))
        (propertize line1 'face '(:weight bold :foreground "#5fafff")))))

  (defun tdw/render-view-banner (total)
    "Insert (or refresh) the 2-line view banner at the top of the agenda buffer.
The view is identified by the sticky agenda buffer name `*Org Agenda(KEY)*':
i=Inbox, S=Someday, u=Unordered, g=Ordered (org-gtd-engage's default key).
The view `name' is only the agenda-dispatcher description and is NOT present in
the buffer body, so we key off the buffer name (which survives refreshes)
rather than searching for title text.  Idempotent: strips a prior banner first."
    (let* ((bn (buffer-name))
           (key (and (string-match "\\*Org Agenda(\\(.+\\))\\*" bn)
                     (match-string 1 bn)))
           (spec (cond ((equal key "i") '("Inbox" . "📥"))
                       ((equal key "S") '("Someday" . "💤"))
                       ((equal key "u") '("Unordered" . "📋"))
                       ((equal key "g") '("Ordered" . "🎯"))))
           (inhibit-read-only t))
      (when spec
        (goto-char (point-min))
        ;; Strip a previously-inserted banner (rule line, plus the effort
        ;; line when present — Inbox/Someday banners have no effort line).
        ;; The rule line always contains `═' even if it doesn't start with it.
        (when (looking-at ".*═")
          (forward-line 1)
          (when (looking-at "[ \t]*Total Estimated Effort:")
            (forward-line 1))
          (delete-region (point-min) (point))
          (goto-char (point-min)))
        (insert (tdw/format-view-banner (car spec) (cdr spec) total nil)
                "\n"))))

  (defun tdw/update-sanity-view-headers ()
    "Update headers in the current agenda buffer with fresh effort totals and warnings."
    (let ((inhibit-read-only t))
      (save-excursion
        (goto-char (point-min))
        ;; Detect if we are in Ordered View
        (when (or (re-search-forward "Ordered View" nil t)
                  (re-search-forward "GTD Engage View" nil t)
                  (re-search-forward "Unordered View" nil t)
                  (re-search-forward "Someday View" nil t)
                  (re-search-forward "Inbox View" nil t)
                  (re-search-forward "🚨🚨 P0" nil t)
                  (re-search-forward "🔔🔔 Bells Ringing" nil t)
                  (re-search-forward "🥎 🥎 On Deck" nil t)
                  (re-search-forward "💭 💭 Recently Considered" nil t)
                  (re-search-forward "🕐 Paused" nil t)
                  (re-search-forward "🪨🪨 Big Rocks" nil t))
          (pcase-let ((`(,total ,effort-p0 ,effort-bell ,effort-deck ,effort-big ,effort-other ,n-unest ,effort-considered ,effort-paused) (tdw/get-sanity-effort-totals)))
            ;; Render the prominent 2-line banner for the 4 banner-able views
            ;; (Inbox / Someday / Unordered / Ordered, incl. the initial
            ;; "GTD Engage View" name). Replaces the old per-view title rewrites
            ;; and relabels the effort line to "Total Estimated Effort:".
            (tdw/render-view-banner total)
            ;; Section-header effort totals — skipped for Inbox ("i") and
            ;; Someday ("S"), which are effort-free views.
            (let* ((bn (buffer-name))
                   (vkey (and (string-match "\\*Org Agenda(\\(.+\\))\\*" bn)
                              (match-string 1 bn)))
                   (show-effort (not (member vkey '("i" "S")))))
              (when show-effort
            ;; P0 effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🚨🚨 P0\\( ([^)]*)\\)?\\)" nil t)
              (replace-match (format "🚨🚨 P0 (%s)" effort-p0) t t nil 1))
            ;; Bells Ringing effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🔔🔔 Bells Ringing\\( ([^)]*)\\)?\\)" nil t)
              (replace-match (format "🔔🔔 Bells Ringing (%s)" effort-bell) t t nil 1))
            ;; On Deck effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🥎 🥎 On Deck\\( ([^)]*)\\)?\\)" nil t)
              (replace-match (format "🥎 🥎 On Deck (%s)" effort-deck) t t nil 1))
            ;; Recently Considered effort
            (goto-char (point-min))
            (while (re-search-forward "\\(💭 💭 Recently Considered\\( ([^)]*)\\)?\\)" nil t)
              (replace-match (format "💭 💭 Recently Considered (%s)" effort-considered) t t nil 1))
            ;; Paused effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🕐 Paused\\( ([^)]*)\\)?\\)" nil t)
              (replace-match (format "🕐 Paused (%s)" effort-paused) t t nil 1))
            ;; Big Rocks effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🪨🪨 Big Rocks\\( ([^)]*)\\)? ?🪨?🪨?\\)" nil t)
              (replace-match (format "🪨🪨 Big Rocks (%s)" effort-big) t t nil 1))
            ;; Other Rocks effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🪨 Other Rocks\\( ([^)]*)\\)? ?🪨?\\)" nil t)
              (replace-match (format "🪨 Other Rocks (%s)" effort-other) t t nil 1))
            ;; NOTE: Unestimated count is intentionally NOT updated here.
            ;; Each view (Someday, Ordered) computes its own accurate count
            ;; at construction time using tdw/get-someday-unestimated-count or
            ;; tdw/get-actions-unestimated-count. Patching it here with n-unest
            ;; (which only counts NEXT/TODO items, not Someday items) would corrupt
            ;; the Someday view's count.
            ))
            ;; Strip NEXT/TODO/WAIT keywords from all lines (case-sensitive
            ;; to avoid corrupting task titles containing "next", "Next", etc.)
            (let ((case-fold-search nil))
              (goto-char (point-min))
              (while (re-search-forward "\\bNEXT \\|\\bTODO \\|\\bWAIT " nil t)
                (replace-match ""))
              ;; Relocate visible keywords (SOMEDAY/DONE/CNCL) from the start
              ;; of the headline text to just after the ' — ' separator, so
              ;; they align like the in-place inserts from the todo advice.
              ;; The padding moves after the keyword, keeping headlines put.
              (goto-char (point-min))
              (while (re-search-forward "\\( — \\)\\( +\\)\\(SOMEDAY\\|DONE\\|CNCL\\) " nil t)
                (replace-match "\\1\\3 \\2")))
            ;; Strip trailing org tags (:tag1:tag2:) and empty :: from all lines,
            ;; but KEEP the tgl_* project tag — it is the one tag we want shown on
            ;; the right (`org-agenda-hide-tags-regexp' has already removed every
            ;; non-tgl tag by the time this runs).
            (goto-char (point-min))
            (while (re-search-forward " +:[a-zA-Z0-9_:]+: *$" nil t)
              (unless (string-match-p ":tgl_" (match-string 0))
                (replace-match "")))
            ;; Inject capacity warnings
            (let ((warnings (tdw/sanity-view-warnings)))
              (dolist (w warnings)
                (goto-char (point-min))
                (when (re-search-forward (regexp-quote (car w)) nil t)
                  (end-of-line)
                  (insert (cdr w)))))
            ;; Strip priority cookies [#X] in Unordered View only
            (when tdw--is-unordered-view
              (goto-char (point-min))
              (let ((cookie-re (concat " ?" (regexp-quote "[") "#[A-Z]" (regexp-quote "]") " ?")))
                (while (re-search-forward cookie-re nil t)
                  (replace-match " ")))))))))

  (defun tdw/insert-tgl-subheaders ()
    "Insert visual sub-headers within the Toggl Projects section.
Collects all task lines, groups them by tgl_* tag, rewrites the
section with grouped lines, and inserts readable dividers."
    (when (get-text-property (point-min) 'org-agenda-type)
      (save-excursion
        (goto-char (point-min))
        (when (re-search-forward "^Toggl Projects$" nil t)
          (forward-line 1)
          (let ((inhibit-read-only t)
                (section-start (point))
                (section-end nil)
                (entries nil))
            ;; Find where the next section header starts (or end of buffer)
            (save-excursion
              (if (re-search-forward "^[^\n ].+$" nil t)
                  (setq section-end (line-beginning-position))
                (setq section-end (point-max))))
            ;; Pass 1: collect all lines with their tgl tag and text properties
            (while (and (< (point) section-end) (not (eobp)))
              (let* ((line-start (line-beginning-position))
                     (line-end (min (1+ (line-end-position)) section-end))
                     (line-text (buffer-substring line-start line-end))
                     (tags (get-text-property (point) 'tags))
                     (tgl-tag nil))
                (when tags
                  (let ((found (cl-find-if
                                (lambda (tg)
                                  (let ((s (if (symbolp tg) (symbol-name tg) tg)))
                                    (and (stringp s) (string-prefix-p "tgl_" s))))
                                tags)))
                    (when found
                      (setq tgl-tag (if (symbolp found) (symbol-name found) found)))))
                (push (list (or tgl-tag "zzz_no_tag") line-text) entries))
              (forward-line 1))
            ;; Pass 2: sort entries by tgl tag, then rewrite the section
            (setq entries (sort (nreverse entries)
                                (lambda (a b) (string< (car a) (car b)))))
            (delete-region section-start section-end)
            (goto-char section-start)
            (let ((current-tag nil))
              (dolist (entry entries)
                (let ((tag (car entry))
                      (text (cadr entry)))
                  (unless (equal tag current-tag)
                    (setq current-tag tag)
                    (unless (equal tag "zzz_no_tag")
                      (let* ((readable (mapconcat #'capitalize
                                                  (split-string (substring tag 4) "_")
                                                  " "))
                             (header (concat "  — " readable " —\n")))

  ;;; --- Gap total in Today's Schedule header ---
  ;; Computes unaccounted time between schedule entries and appends
  ;; "| Gap: H:MM" to the Today's Schedule header.

  (defun tdw/compute-gap-total ()
    "Compute gap minutes from the Today's Schedule block in the agenda buffer.
Scans time-grid entries (HH:MM lines) between the schedule header and the
next section separator, collecting start times from timesheet.org and gcal.org
entries. Returns the total unaccounted gap in minutes."
    (save-excursion
      (goto-char (point-min))
      (when (re-search-forward "📅 Today's Schedule" nil t)
        (let ((section-end (save-excursion
                             (if (re-search-forward "^=+" nil t)
                                 (line-beginning-position)
                               (point-max))))
              (entries nil)
              (now-minutes (+ (* (string-to-number (format-time-string "%H")) 60)
                              (string-to-number (format-time-string "%M")))))
          ;; Collect (start-min . end-min) pairs from agenda time entries.
          ;; Agenda time entries look like: "   10:00...... 0:30  Meeting Name"
          ;; where the time is the start and the effort (if present) gives duration.
          (while (and (< (point) section-end)
                      (re-search-forward
                       "^\\s-+\\([0-9]\\{1,2\\}\\):\\([0-9]\\{2\\}\\)\\.\\{4,\\}\\s-+\\(?:\\([0-9]+\\):\\([0-9]\\{2\\}\\)\\)?"
                       section-end t))
            (let* ((sh (string-to-number (match-string 1)))
                   (sm (string-to-number (match-string 2)))
                   (start-min (+ (* sh 60) sm))
                   (dur-h (when (match-string 3) (string-to-number (match-string 3))))
                   (dur-m (when (match-string 4) (string-to-number (match-string 4))))
                   (dur (if (and dur-h dur-m) (+ (* dur-h 60) dur-m) 30))
                   (end-min (+ start-min dur)))
              (push (cons start-min end-min) entries)))
          ;; Also collect entries from Timesheet category lines (no dots)
          ;; These appear as: "  Timesheet:  07:00-07:15 Task Name"
          ;; or via the standard agenda format with a time range.
          (goto-char (point-min))
          (when (re-search-forward "📅 Today's Schedule" nil t)
            (while (and (< (point) section-end)
                        (re-search-forward
                         "Timesheet:\\s-+\\([0-9]\\{1,2\\}\\):\\([0-9]\\{2\\}\\)-\\([0-9]\\{1,2\\}\\):\\([0-9]\\{2\\}\\)"
                         section-end t))
              (let* ((sh (string-to-number (match-string 1)))
                     (sm (string-to-number (match-string 2)))
                     (eh (string-to-number (match-string 3)))
                     (em (string-to-number (match-string 4)))
                     (start-min (+ (* sh 60) sm))
                     (end-min (+ (* eh 60) em)))
                (push (cons start-min end-min) entries))))
          ;; Compute gap: sort entries, walk timeline, sum unaccounted gaps
          (when entries
            (setq entries (sort entries (lambda (a b) (< (car a) (car b)))))
            (let ((gap 0)
                  (cursor (caar entries)))
              (dolist (e entries)
                (when (> (car e) cursor)
                  (setq gap (+ gap (- (car e) cursor))))
                (when (> (cdr e) cursor)
                  (setq cursor (cdr e))))
              ;; Gap from last entry to now
              (when (> now-minutes cursor)
                (setq gap (+ gap (- now-minutes cursor))))
              gap))))))

  (defun tdw/insert-gap-total ()
    "Append gap total to the Today's Schedule header."
    (let ((gap (tdw/compute-gap-total)))
      (when (and gap (> gap 0))
        (let ((inhibit-read-only t)
              (gap-str (tdw/format-minutes gap)))
          (save-excursion
            (goto-char (point-min))

              (replace-match (format "\\1 | Gap: %s" gap-str) t nil)))))))

  ;;; --- Currently-running clock under the schedule header ---
  ;; The in-memory `org-clock-*' state is per-Emacs-process and goes stale: if
  ;; the clock is running in a different Emacs than the agenda daemon, or after a
  ;; clock-out, `org-clock-get-clocked-time' returns a frozen/garbage value. So
  ;; we read GROUND TRUTH instead: scan the GTD org files for an OPEN `CLOCK:'
  ;; line (one with no closing `--end') and compute elapsed from its timestamp at
  ;; render time. This recomputes fresh on every `r', works regardless of which
  ;; process started the clock, and shows nothing when no clock is open.

  (defun tdw/clean-heading (h)
    "Strip TODO keyword, priority cookie, and trailing tags from heading H."
    (let ((s h))
      (setq s (replace-regexp-in-string
               "^\\(TODO\\|NEXT\\|WAIT\\|DONE\\|CNCL\\|SOMEDAY\\)[ \t]+" "" s))
      (setq s (replace-regexp-in-string "\\[#.\\][ \t]*" "" s))
      (setq s (replace-regexp-in-string "[ \t]+:[[:alnum:]_@#%:]+:[ \t]*$" "" s))
      (string-trim s)))

  (defun tdw/active-clock-info ()
    "Return (HEADING . START-TIME) for the most recent open CLOCK, or nil.
Scans all org-gtd-tasks.org files (primary + context repos) and timesheet.org
for a `CLOCK: [ts]' line with no closing `--', picking the latest. Reflects the
true running clock across files and processes."
    (let ((files (append (tdw/all-tasks-files)
                         (list (expand-file-name "timesheet.org" stag-org-gtd-directory))))
          (best nil))
      (dolist (file files best)
        (when (file-exists-p file)
          (with-temp-buffer
            (insert-file-contents file)
            (goto-char (point-min))
            (while (re-search-forward "^[ \t]*CLOCK: \\(\\[[^]]+\\]\\)[ \t]*$" nil t)
              (let* ((start (org-time-string-to-time (match-string 1)))
                     (heading (save-excursion
                                (when (re-search-backward "^\\*+ +\\(.*\\)$" nil t)
                                  (match-string 1)))))
                (when (or (null best) (time-less-p (cdr best) start))
                  (setq best (cons (tdw/clean-heading (or heading "")) start))))))))))

  (defun tdw/agenda-adjust-effort (&optional delta)
    "Adjust effort of agenda item at point by DELTA minutes (defaults to 15)."
    (interactive "P")
    (require 'org-duration)
    (let ((delta-mins (cond ((numberp delta) (round delta))
                            ((and (listp delta) (numberp (car delta))) (round (car delta)))
                            (t 15))))
      (if-let ((marker (org-get-at-bol 'org-marker)))
          (let* ((current (org-with-point-at marker (org-entry-get nil "Effort")))
                 (current-mins (condition-case nil
                                   (if (and (stringp current) (not (string-empty-p current)))
                                       (round (org-duration-to-minutes current))
                                     0)
                                 (error 0)))
                 (new-mins (max 0 (+ current-mins delta-mins)))
                 (new-effort (org-duration-from-minutes new-mins)))
            (org-with-point-at marker
              (org-entry-put nil "Effort" new-effort))
            ;; Update effort prefix in-place instead of full refresh
            ;; Restrict search to prefix area (before " — ") to avoid
            ;; matching habit graph brackets or scheduled times
            (let ((inhibit-read-only t))
              (save-excursion
                (beginning-of-line)
                (let ((prefix-end (save-excursion
                                    (if (re-search-forward " — " (line-end-position) t)
                                        (match-beginning 0)
                                      (line-end-position)))))
                  (when (re-search-forward "[0-9]+:[0-9]+" prefix-end t)
                    (replace-match new-effort)))))
            (message "Effort: %s" new-effort))
        (message "No task at point"))))

                          (unless found-sep

  (defun tdw/agenda--refresh-prefix (hdmarker buffer)
    "Re-resolve the agenda prefix annotation for the line at point, in place.
Refreshes the tier glyph (🚨/🔔/🥎/💭/🕐) after a tier change without rebuilding
the whole agenda. The tier glyph cluster sits immediately before the ' — '
separator, followed by alignment padding; replace the cluster and re-pad to
keep columns stable."
    (let ((inhibit-read-only t))
      (save-excursion
        (beginning-of-line)
        (cond
         ((re-search-forward "\\([[:nonascii:]]*\\)\\( — *\\)" (line-end-position) t)
          (let* ((mbeg (match-beginning 0))
                 (mend (match-end 0))
                 (oldw (string-width (buffer-substring mbeg mend)))
                 (new (with-current-buffer buffer
                        (org-with-wide-buffer
                         (goto-char hdmarker)
                         (org-gtd-agenda--resolve-tier))))
                 (padw (max 0 (- oldw (string-width new) 3)))) ; 3 = " — "
            (delete-region mbeg mend)
            (goto-char mbeg)
            (insert new " — " (make-string padw ?\s))))))))

    "Set priority tier TAG on the agenda task at point, clearing every other tier.
TAG is one of `tdw--tier-tags'. If the task already carries TAG, remove it instead
\(back to no tier). Tiers are mutually exclusive: a task holds at most one."
    (let* ((hdmarker (or (org-get-at-bol 'org-hd-marker)
                         (org-agenda-error)))
           (buffer (marker-buffer hdmarker))

      (with-current-buffer buffer
        (org-with-wide-buffer
         (goto-char hdmarker)
         (let ((tags (org-get-tags nil t)))

             (when (member tier tags)
               (org-toggle-tag tier 'off))))
         ;; Re-pressing the active tier clears to nothing; else apply the choice
         (unless had-tag
           (org-toggle-tag tag 'on))))
      (tdw/agenda--refresh-prefix hdmarker buffer)
      (message "%s %s" (if had-tag "Cleared" "Set") tag)))

  ;;; ——— Score Computation ———

  (defconst tdw--level-weights
    '(("wh" . 8) ("vh" . 5) ("h" . 3) ("sh" . 2) ("m" . 1) ("l" . 0))
    "Urgency/impact level abbreviations to numeric weight, highest first.
Single source of truth for both the level vocabulary and the scoring.")

  (defun tdw/score-value-for-level (level)
    "Map urgency/impact level abbreviation to its numeric weight (unknown -> 0)."
    (or (cdr (assoc level tdw--level-weights)) 0))

  (defun tdw/compute-score-at-point ()
    "Compute floor((2*urg + imp) / effort-tier) at point. Returns integer.
Effort tiers: 0-5min=1, 6-15min=2, 16-60min=3, 60+=4.
Higher score = more value per unit of effort."
    (let* ((tags (org-get-tags nil t))
           (urg-level (or (cl-find-if (lambda (lvl)
                                        (member (format "%s_urgency" lvl) tags))
                                      (mapcar #'car tdw--level-weights))
                          "l"))
           (imp-level (or (cl-find-if (lambda (lvl)
                                        (member (format "%s_impact" lvl) tags))
                                      (mapcar #'car tdw--level-weights))
                          "l"))
           (urg (tdw/score-value-for-level urg-level))
           (imp (tdw/score-value-for-level imp-level))
           ;; Untimed or zero-effort items default to the smallest tier (5 min).
           (effort-mins (let ((m (tdw/get-effort-minutes (org-entry-get (point) "EFFORT"))))
                          (if (> m 0) m 5)))
           (tier (cond ((<= effort-mins 5) 1)
                       ((<= effort-mins 15) 2)
                       ((<= effort-mins 60) 3)
                       (t 4)))
           (raw (+ (* 2 urg) imp)))
      (floor (/ (float raw) tier))))

           (hdl (org-get-heading t t t t)))
      (if (and is-habit (not done)
               (cl-some (lambda (name) (string-match-p name hdl))
                        tdw--ritual-habits))
          nil
        (outline-next-heading))))

  (defun tdw/skip-unless-non-ritual-habit ()
    "Skip unless this is an undone habit that is NOT a daily ritual."
    (let* ((dominated (org-entry-get (point) "STYLE"))
           (is-habit (string= (or dominated "") "habit"))
           (done (org-entry-is-done-p))
           (hdl (org-get-heading t t t t)))
      (if (and is-habit (not done)
               (not (cl-some (lambda (name) (string-match-p name hdl))
                             tdw--ritual-habits)))
          nil
        (outline-next-heading))))

  (defun tdw/skip-unless-tickler-due ()
    "Skip unless this is an incubated item scheduled for today or earlier."
    (let* ((scheduled (org-get-scheduled-time (point)))
           (today (org-today)))
      (if (and scheduled
               (not (org-entry-is-done-p))
               (<= (org-time-string-to-absolute
                    (format-time-string "%Y-%m-%d" scheduled))
                   today))
          nil
        (org-entry-end-position))))

  (defun tdw--score-effort-at-point ()
    "Return (score effort-mins on-p0-p done-p gc-next-p flyby-p) at point for skip functions."
    (let* ((tags (org-get-tags nil t))
           (score (tdw/compute-score-at-point))
           (effort-str (org-entry-get (point) "EFFORT"))
           (effort-mins (and effort-str (tdw/get-effort-minutes effort-str))))
      (list score effort-mins (member "p0" tags) (org-entry-is-done-p) (member "gc_next" tags) (member "flyby" tags))))

  (defun tdw/skip-unless-unestimated ()
    "Skip unless score < `tdw--big-rock-threshold' and the entry is unestimated."
    (let* ((end (org-entry-end-position))
           (info (tdw--score-effort-at-point))
           (score (nth 0 info))
           (done (nth 3 info)))
      (if (and (not done) (< score tdw--big-rock-threshold)
               (tdw/unestimated-p))
          nil
        end)))

  (defun tdw/skip-cncl-globally ()
    "Skip CNCL (canceled) or DONE entries in ALL agenda views.
The custom views are built from `tags' searches on the ORG_GTD property,
which match entries regardless of TODO state — so canceled and done items
kept showing in Inbox-type blocks.  Installed via
`org-agenda-skip-function-global', which org consults in addition to any
per-block `org-agenda-skip-function'."
    (let ((state (org-get-todo-state)))
      (when (or (string= state "CNCL")
                (string= state "DONE"))
        (org-entry-end-position))))
  (setq org-agenda-skip-function-global #'tdw/skip-cncl-globally)

(defun tdw/startup-single-frame ()
  "Automatically open Unordered View on startup."
  ;; Only render views when we have a real GUI frame (non-daemon).
  ;; In daemon mode, no GUI frame exists at startup — user triggers
  ;; views manually with C-c d a / C-c d u after connecting.
  (unless (daemonp)
    (tdw-unordered-view "u")
    (delete-other-windows)))
