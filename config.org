** Auto-saving
I keep losing work, so let's make Emacs auto-save frequently. We'll use the `super-save` package to save buffers when we switch buffers or windows, plus an aggressive automatic save on idle or losing focus.
#+begin_src emacs-lisp
  (use-package super-save
    :ensure t
    :pin "melpa"
    :config
    ;; Save on window/buffer movements
    (add-to-list 'super-save-triggers 'other-window)
    (add-to-list 'super-save-triggers 'windmove-up)
    (add-to-list 'super-save-triggers 'windmove-down)
    (add-to-list 'super-save-triggers 'windmove-left)
    (add-to-list 'super-save-triggers 'windmove-right)
    
    (super-save-mode +1)
    (setq super-save-auto-save-when-idle t)
    (setq super-save-idle-duration 1)) ;; save after 1 second of idle

  ;; Also save unconditionally when Emacs loses focus
  (add-hook 'focus-out-hook (lambda () (save-some-buffers t)))
#+end_src

#+BEGIN_SRC emacs-lisp
  (defun stag-what-face (pos)
    (interactive "d")
    (let* ((local-hl-p hl-line-mode)
           (global-hl-p global-hl-line-mode)
           (off (if local-hl-p (hl-line-mode -1)))
           (off (if global-hl-p (global-hl-line-mode -1)))
           (face (or (get-char-property pos 'read-face-name)
                     (get-char-property pos 'face))))
      (if face (message "Face: %s" face)
        (message "No face at %d" pos))
      (if local-hl-p (hl-line-mode))
      (if global-hl-p (global-hl-line-mode))))

* GTD (Getting Things Done)
** Priority and tag inheritance for projects
#+BEGIN_SRC emacs-lisp
  ;;; org-gtd priority and tag inheritance for projects

  (defgroup my/org-gtd-inherit nil
    "Customization for org-gtd priority and tag inheritance."
    :group 'org-gtd
    :prefix "my/org-gtd-inherit-")

  (defcustom my/org-gtd-inherit-priority t
    "Whether to inherit priority from parent project.
When non-nil, project tasks automatically inherit the priority
from their parent project heading."
    :type 'boolean
    :group 'my/org-gtd-inherit)

  (defcustom my/org-gtd-inherit-tags t
    "Whether to inherit tags from parent project.
When non-nil, project tasks automatically inherit the tags
from their parent project heading."
    :type 'boolean
    :group 'my/org-gtd-inherit)

  (defun my/org-gtd-inherit--get-parent-metadata ()
    "Get priority and tags from parent heading.
Returns plist with :priority and :tags, or nil if no parent.
Handles buffer narrowing automatically."
    (save-excursion
      (save-restriction
        (widen)
        (when (org-up-heading-safe) ;; it's a dag so have to find the project heading for the org-gtd project property
          (list :priority (org-entry-get (point) "PRIORITY")
                :tags (org-get-tags nil t))))))

  (defun my/org-gtd-inherit--apply-metadata (metadata)
    "Apply METADATA plist (:priority and :tags) to current heading.
Only applies values that are configured for inheritance via customize vars."
    (when-let ((priority (and my/org-gtd-inherit-priority

  (defun tdw/get-effort-minutes (effort-str)
    "Convert EFFORT-STR (H:MM) to minutes."
    (when (and effort-str (string-match "\\([0-9]+\\):\\([0-9]+\\)" effort-str))
      (+ (* 60 (string-to-number (match-string 1 effort-str)))
         (string-to-number (match-string 2 effort-str)))))

  (defun tdw/get-subtree-effort-minutes ()
    "Sum effort of current entry and all its children in minutes."
    (let ((total 0))
      (save-excursion
        (org-back-to-heading t)
        (let ((effort (org-entry-get (point) "EFFORT")))
          (when effort (setq total (tdw/get-effort-minutes effort))))
        (let ((end (save-excursion (org-end-of-subtree t) (point))))
          (outline-next-heading)
          (while (and (< (point) end) (not (eobp)))
            (let ((effort (org-entry-get (point) "EFFORT")))
              (when effort
                (setq total (+ total (or (tdw/get-effort-minutes effort) 0)))))
            (outline-next-heading))))
      total))

                              (t (setq mins-other (+ mins-other effort))))))))
                nil nil))))))

        (list (funcall fmt (+ mins-fire mins-big mins-quick mins-other))

  (defun tdw/sanity-view-warnings ()
    "Return alist of (SECTION . WARNING-STRING) for capacity violations."
    (let ((n-fire 0) (n-vh-urg 0) (n-h-urg 0) (n-vh-imp 0) (n-h-imp 0)
          (warnings nil))
      (dolist (file (org-agenda-files))
        (let ((expanded-file (expand-file-name file)))
          (when (file-exists-p expanded-file)
            (with-current-buffer (find-file-noselect expanded-file)
              (org-with-wide-buffer
               (goto-char (point-min))
               (org-map-entries
                (lambda ()
                  (let ((todo (org-get-todo-state))
                        (tags (org-get-tags nil t)))
                    (when (member todo '("NEXT" "TODO"))
                      (when (member "on_fire" tags) (cl-incf n-fire))
                      (when (member "vh_urgency" tags) (cl-incf n-vh-urg))
                      (when (member "h_urgency" tags) (cl-incf n-h-urg))
                      (when (member "vh_impact" tags) (cl-incf n-vh-imp))
                      (when (member "h_impact" tags) (cl-incf n-h-imp)))))
                nil nil))))))
      (if (> n-fire 5)
          (push (cons "🔥🔥 On Fire"
                       (format " *** WARNING: %d on-fire items — remove %d *** 🔥🔥" n-fire (- n-fire 5)))
                warnings)
        (push (cons "🔥🔥 On Fire"
                     " [blocking someone | due <24h | big commitment at risk] 🔥🔥")
              warnings))
      (if (or (> n-vh-urg 1) (> n-h-urg 2) (> n-vh-imp 1) (> n-h-imp 2))
          (let ((parts nil))
            (when (> n-vh-urg 1) (push (format "%d vh_urgency — remove %d" n-vh-urg (- n-vh-urg 1)) parts))
            (when (> n-h-urg 2) (push (format "%d h_urgency — remove %d" n-h-urg (- n-h-urg 2)) parts))
            (when (> n-vh-imp 1) (push (format "%d vh_impact — remove %d" n-vh-imp (- n-vh-imp 1)) parts))
            (when (> n-h-imp 2) (push (format "%d h_impact — remove %d" n-h-imp (- n-h-imp 2)) parts))
            (push (cons "🪨🪨 Big Rocks"
                         (format " *** WARNING: %s *** 🪨🪨" (string-join (nreverse parts) ", ")))
                  warnings))
        (push (cons "🪨🪨 Big Rocks" " 🪨🪨") warnings))
      (push (cons "💨💨 Quick Wins" " 💨💨") warnings)
      (push (cons "🪨 Other Rocks" " 🪨") warnings)
      warnings))

  (defun tdw/update-sanity-view-headers ()
    "Update headers in the current agenda buffer with fresh effort totals and warnings."
    (let ((inhibit-read-only t))
      (save-excursion
        (goto-char (point-min))
        ;; Detect if we are in Sanity/Decision View
        (when (or (re-search-forward "Sanity View" nil t)
                  (re-search-forward "Decision View" nil t)
                  (re-search-forward "🔥🔥 On Fire" nil t)
                  (re-search-forward "🪨🪨 Big Rocks" nil t)
                  (re-search-forward "Hot Tickets" nil t)
                  (re-search-forward "For Today" nil t)
                  (re-search-forward "Priority C" nil t)
                  (re-search-forward "Priority D" nil t))
          (pcase-let ((`(,total ,effort-fire ,effort-big ,effort-quick ,effort-other ,n-unest) (tdw/get-sanity-effort-totals)))
            ;; Update Sanity View total
            (goto-char (point-min))
            (while (re-search-forward "\\(Sanity View\\( (Total Effort: [^)]*)\\)?\\)" nil t)
              (replace-match (format "Sanity View (Total Effort: %s)" total) t t nil 1))
            ;; Decision View total (kept for backward compat)
            (goto-char (point-min))
            (while (re-search-forward "\\(Decision View\\( (Total Effort: [^)]*)\\)?\\)" nil t)
              (replace-match (format "Decision View (Total Effort: %s)" total) t t nil 1))
            ;; On Fire effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🔥🔥 On Fire\\( ([^)]*)\\)?\\)" nil t)
              (replace-match (format "🔥🔥 On Fire (%s)" effort-fire) t t nil 1))
            ;; Big Rocks effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🪨🪨 Big Rocks\\( ([^)]*)\\)? ?🪨?🪨?\\)" nil t)
              (replace-match (format "🪨🪨 Big Rocks (%s)" effort-big) t t nil 1))
            ;; Quick Wins effort
            (goto-char (point-min))
            (while (re-search-forward "\\(💨💨 Quick Wins\\( ([^)]*)\\)? ?💨?💨?\\)" nil t)
              (replace-match (format "💨💨 Quick Wins (%s)" effort-quick) t t nil 1))
            ;; Other Rocks effort
            (goto-char (point-min))
            (while (re-search-forward "\\(🪨 Other Rocks\\( ([^)]*)\\)? ?🪨?\\)" nil t)
              (replace-match (format "🪨 Other Rocks (%s)" effort-other) t t nil 1))
            ;; Hot Tickets / For Today / Priority C / Priority D (Decision View compat)
            (goto-char (point-min))
            (while (re-search-forward "\\(Hot Tickets\\( (Priority A - [^)]*)\\)?\\)" nil t)
              (replace-match (format "Hot Tickets (Priority A - %s)" effort-fire) t t nil 1))
            ;; Unestimated count
            (goto-char (point-min))
            (while (re-search-forward "\\(Unestimated\\( ([^)]*)\\)?\\)" nil t)
              (replace-match (format "Unestimated (%d items)" n-unest) t t nil 1))
            (goto-char (point-min))
            (while (re-search-forward "\\(For Today\\( (Priority B - [^)]*)\\)?\\)" nil t)
              (replace-match (format "For Today (Priority B - %s)" effort-big) t t nil 1))
            ;; Inject capacity warnings
            (let ((warnings (tdw/sanity-view-warnings)))
              (dolist (w warnings)
                (goto-char (point-min))
                (when (re-search-forward (regexp-quote (car w)) nil t)
                  (end-of-line)
                  (insert (cdr w))))))))))

  (defun tdw/agenda-cycle-tag (prefix levels)
    "Cycle a tag with PREFIX through LEVELS on the task at point.
PREFIX is e.g. \"urgency\" or \"impact\".
LEVELS is a list like (\"l\" \"m\" \"h\" \"vh\")."
    (let* ((hdmarker (or (org-get-at-bol 'org-hd-marker)
                         (org-agenda-error)))
           (buffer (marker-buffer hdmarker))
           (pos (marker-position hdmarker))
           current-level current-tag next-level next-tag)
      (with-current-buffer buffer
        (widen)
        (goto-char pos)
        (let ((tags (org-get-tags nil t)))
          ;; Find current level
          (setq current-level
                (cl-find-if (lambda (lvl)
                              (member (format "%s_%s" lvl prefix) tags))
                            levels))
          (setq current-tag (when current-level
                              (format "%s_%s" current-level prefix)))
          ;; Determine next level (cycle)
          (setq next-level
                (if current-level
                    (let ((idx (cl-position current-level levels :test #'equal)))
                      (nth (mod (1+ idx) (length levels)) levels))
                  (car levels)))
          (setq next-tag (format "%s_%s" next-level prefix))
          ;; Remove old, add new
          (when current-tag (org-toggle-tag current-tag 'off))
          (org-toggle-tag next-tag 'on)))
      ;; Update prefix in-place instead of full refresh
      (let ((inhibit-read-only t)
            (new-score (with-current-buffer buffer
                         (org-with-wide-buffer
                          (goto-char pos)
                          (org-gtd-agenda--resolve-score)))))
        (save-excursion
          (beginning-of-line)
          (when (re-search-forward "\\(🔥 \\)?U:[^ ]* I:[^ ]*" (line-end-position) t)
            (replace-match new-score))))
      (message "Set %s → %s" prefix next-level)))

  (defun tdw/agenda-score-at-marker (marker)
    "Compute urgency/impact score at MARKER: 2*urgency + impact.
vh=3, h=2, m=1, l=0."
    (or (org-with-point-at marker
          (let* ((tags (org-get-tags nil t))
                 (urg (cond ((member "vh_urgency" tags) 3)
                            ((member "h_urgency" tags) 2)
                            ((member "m_urgency" tags) 1)
                            (t 0)))
                 (imp (cond ((member "vh_impact" tags) 3)
                            ((member "h_impact" tags) 2)
                            ((member "m_impact" tags) 1)
                            (t 0))))
            (+ (* 2 urg) imp)))
        0))

  (defun tdw/agenda-cmp-score (a b)
    "Compare agenda entries A and B by urgency/impact score.
Returns +1 if A should sort first (higher score), -1 if B, nil if equal."
    (let* ((ma (or (get-text-property 1 'org-marker a)
                   (get-text-property 1 'org-hd-marker a)))
           (mb (or (get-text-property 1 'org-marker b)
                   (get-text-property 1 'org-hd-marker b)))
           (sa (if ma (tdw/agenda-score-at-marker ma) 0))
           (sb (if mb (tdw/agenda-score-at-marker mb) 0)))
      (cond ((> sa sb) +1)
            ((< sa sb) -1)
            (t nil))))
